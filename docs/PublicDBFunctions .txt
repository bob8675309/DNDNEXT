END IF;

END LOOP;
-- 2) PLAN NEXT LEG FOR MERCHANTS WHO HAVE FINISHED RESTING

FOR m IN SELECT * FROM public.merchants WHERE state IN ('resting','moving') -- eligible to start a leg AND route_id IS NOT NULL AND (rest_until IS NULL OR rest_until <= p_now) LOOP ----------------------------------------------------------------- -- Route metadata: route_type, is_loop, max_seq ----------------------------------------------------------------- SELECT r.route_type, r.is_loop, ( SELECT max(rp2.seq) FROM public.map_route_points rp2 WHERE rp2.route_id = m.route_id ) AS max_seq INTO v_route_type, v_is_loop, v_max_seq FROM public.map_routes r WHERE r.id = m.route_id;*

-- If route has no points, skip this merchant IF v_max_seq IS NULL THEN CONTINUE; END IF;
SQL Query

-----------------------------------------------------------------
-- Current point on this route
-----------------------------------------------------------------
SELECT *
INTO  curr_pt
FROM  public.map_route_points rp
WHERE rp.route_id = m.route_id
  AND rp.seq = COALESCE(m.route_point_seq, m.current_point_seq, 1)
LIMIT 1;

IF curr_pt.id IS NULL THEN
  CONTINUE;
END IF;


-----------------------------------------------------------------
-- 80%: targeted travel on trade (teal) routes
-- 20%: local random walk (or always random on non-trade)
-----------------------------------------------------------------
IF v_route_type = 'trade' THEN
  v_rand := random();            -- 0..1
  v_use_target := (v_rand < 0.8); -- 80% of the time
ELSE
  v_use_target := FALSE;         -- excursions etc → always random for now
END IF;

v_target_loc := NULL;
v_target_seq := NULL;
v_next_seq   := NULL;

-----------------------------------------------------------------
-- TARGETED LEG (80% of the time on trade routes)
-----------------------------------------------------------------
IF v_use_target THEN
  -- 1) validate existing target_location_id (projected_destination_id)
  v_target_loc := m.projected_destination_id;

  IF v_target_loc IS NOT NULL THEN
    PERFORM 1
    FROM public.map_route_points rp
    WHERE rp.route_id = m.route_id
      AND rp.location_id = v_target_loc
    LIMIT 1;

    IF NOT FOUND THEN
      v_target_loc := NULL;
    END IF;
  END IF;

  -- 2) if no valid target, roll a new one from locations on this route
  IF v_target_loc IS NULL THEN
    SELECT loc_id
    INTO   v_target_loc
    FROM (
      SELECT DISTINCT rp.location_id AS loc_id
      FROM   public.map_route_points rp
      WHERE  rp.route_id = m.route_id
        AND  rp.location_id IS NOT NULL
        AND  (rp.location_id IS DISTINCT FROM curr_pt.location_id)
    ) t
    ORDER BY random()
    LIMIT 1;

    IF v_target_loc IS NULL THEN
      -- no other locations; fall back to local random
      v_use_target := FALSE;
    END IF;
  END IF;
END IF;

-----------------------------------------------------------------
-- If using a target, choose direction around the route toward it
-----------------------------------------------------------------
IF v_use_target THEN
  -- nearest point on this route that represents the target location
  SELECT rp.seq
  INTO   v_target_seq
  FROM   public.map_route_points rp
  WHERE  rp.route_id = m.route_id
    AND  rp.location_id = v_target_loc
  ORDER BY abs(rp.seq - curr_pt.seq)
  LIMIT 1;

  IF v_target_seq IS NULL OR v_target_seq = curr_pt.seq THEN
    -- already at target: stay resting and clear target
    UPDATE public.merchants
    SET projected_destination_id = NULL,
        state = 'resting',
        rest_until = p_now + make_interval(
                        secs => GREATEST(curr_pt.dwell_seconds, 0)
                      )
    WHERE id = m.id;
    CONTINUE;
  END IF;

  v_n := v_max_seq;

  IF v_is_loop THEN
    v_forward  := (v_target_seq - curr_pt.seq + v_n) % v_n;
    v_backward := (curr_pt.seq - v_target_seq + v_n) % v_n;

    IF v_forward <= 0 THEN
      v_dir := 0;
    ELSIF v_forward <= v_backward THEN
      v_dir := 1;
    ELSE
      v_dir := -1;
    END IF;
  ELSE
    IF v_target_seq > curr_pt.seq THEN
      v_dir := 1;
    ELSE
      v_dir := -1;
    END IF;
  END IF;

  IF v_dir = 0 THEN
    CONTINUE;
  END IF;

  IF v_is_loop THEN
    IF v_dir = 1 THEN
      v_next_seq := (curr_pt.seq % v_n) + 1;
    ELSE
      v_next_seq := ((curr_pt.seq + v_n - 2) % v_n) + 1;
    END IF;
  ELSE
    IF v_dir = 1 THEN
      v_next_seq := LEAST(curr_pt.seq + 1, v_n);
    ELSE
      v_next_seq := GREATEST(curr_pt.seq - 1, 1);
    END IF;
  END IF;

ELSE
-----------------------------------------------------------------
-- LOCAL RANDOM LEG (20% of the time, or for non-trade routes)
-----------------------------------------------------------------
  v_n := v_max_seq;
  v_neighbors := ARRAY[]::integer[];

  IF v_is_loop OR curr_pt.seq > 1 THEN
    v_neighbors := array_append(
      v_neighbors,
      CASE WHEN curr_pt.seq > 1 THEN curr_pt.seq - 1 ELSE v_n END
    );
  END IF;

  IF v_is_loop OR curr_pt.seq < v_n THEN
    v_neighbors := array_append(
      v_neighbors,
      CASE WHEN curr_pt.seq < v_n THEN curr_pt.seq + 1 ELSE 1 END
    );
  END IF;

  IF v_neighbors IS NULL OR array_length(v_neighbors, 1) = 0 THEN
    CONTINUE;
  END IF;

  v_idx := floor(random() * array_length(v_neighbors, 1))::integer + 1;
  v_next_seq := v_neighbors[v_idx];
  v_target_loc := NULL;  -- wandering clears long-term goal
END IF;

IF v_next_seq IS NULL THEN
  CONTINUE;
END IF;

-- destination point for the next leg
SELECT *
INTO   to_pt
FROM   public.map_route_points
WHERE  route_id = m.route_id
  AND  seq = v_next_seq
LIMIT 1;

IF to_pt.id IS NULL THEN
  CONTINUE;
END IF;

-----------------------------------------------------------------
-- Schedule the new segment
-----------------------------------------------------------------
v_dist := sqrt( (to_pt.x - curr_pt.x)^2 + (to_pt.y - curr_pt.y)^2 );
v_speed := GREATEST(m.roaming_speed, 0.001);
v_travel_seconds := GREATEST(60.0, v_dist / v_speed);  -- clamp ≥ 60s

UPDATE public.merchants
SET state              = CASE WHEN m.state = 'excursion'
                              THEN 'excursion'
                              ELSE 'moving'
                         END,
    current_point_seq  = curr_pt.seq,
    next_point_seq     = to_pt.seq,
    route_point_seq    = curr_pt.seq,
    segment_started_at = p_now,
    segment_ends_at    = p_now + make_interval(secs => v_travel_seconds),
    rest_until         = NULL,
    route_segment_progress = 0.0,
    x                  = curr_pt.x,
    y                  = curr_pt.y,
    last_moved_at      = p_now,
    projected_destination_id = v_target_loc
WHERE id = m.id;*

END LOOP;

END; $function$

CREATE OR REPLACE FUNCTION public.advance_world_time_v1() RETURNS world_state LANGUAGE plpgsql AS $function$ DECLARE ws public.world_state; real_delta_seconds double precision; game_delta interval; BEGIN SELECT * INTO ws FROM public.world_state WHERE id = 1 FOR UPDATE;*

CREATE OR REPLACE FUNCTION public.buy_from_merchant(p_merchant_id uuid, p_stock_uuid uuid, p_qty integer) RETURNS integer LANGUAGE plpgsql SECURITY DEFINER SET search_path TO 'public' AS $function$ declare buyer uuid := auth.uid(); buy_qty int; price_each numeric; total numeric; cur_gp numeric; stock public.character_stock%rowtype; i int; per_item_payload jsonb; begin if buyer is null then raise exception 'Not signed in'; end if;

buy_qty := greatest(coalesce(p_qty, 1), 1);

-- Lock the stock row select * into stock from public.character_stock where id = p_stock_uuid and character_id = p_merchant_id for update;*

if not found then raise exception 'Stock not found'; end if;

if stock.qty < buy_qty then raise exception 'Not enough stock'; end if;

price_each := coalesce(stock.price_gp, 0); total := price_each * buy_qty;*

-- Ensure wallet row exists and lock it insert into public.player_wallets(user_id, gp) values (buyer, 0) on conflict (user_id) do nothing;

select gp into cur_gp from public.player_wallets where user_id = buyer for update; cur_gp := coalesce(cur_gp, 0);

if cur_gp <> -1 and cur_gp < total then raise exception 'Insufficient funds'; end if;

-- Debit wallet unless infinite if cur_gp <> -1 then update public.player_wallets set gp = cur_gp - total, updated_at = now() where user_id = buyer; end if;

-- Decrement stock by the purchased quantity (do not zero out the whole row) update public.character_stock set qty = qty - buy_qty, updated_at = now() where id = p_stock_uuid;

-- Clean up empty stock rows delete from public.character_stock where id = p_stock_uuid and qty <= 0;

-- Insert inventory rows (inventory_items has no quantity field) per_item_payload := coalesce(stock.card_payload, '{}'::jsonb) || jsonb_build_object( 'purchase_price_gp', price_each, 'purchase_total_gp', total, 'purchase_qty', buy_qty, 'merchant_id', p_merchant_id, 'stock_id', p_stock_uuid );

for i in 1..buy_qty loop insert into public.inventory_items( user_id, item_id, item_name, item_type, item_rarity, item_description, item_weight, item_cost, card_payload, owner_type, owner_id, is_equipped ) values ( buyer, coalesce(stock.item_id, stock.id::text), stock.display_name, (stock.card_payload ->> 'item_type'), (stock.card_payload ->> 'item_rarity'), coalesce(stock.card_payload ->> 'item_description', stock.card_payload ->> 'description'), (stock.card_payload ->> 'item_weight'), (stock.card_payload ->> 'item_cost'), per_item_payload, 'player', buyer::text, false ); end loop;

return buy_qty; end; $function$
